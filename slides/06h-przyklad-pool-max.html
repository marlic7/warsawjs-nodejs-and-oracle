Przykład 3a. Przekroczenie póli (oracledb)

var async    = require('async'),
    oracledb = require('oracledb'),
    cfg  = {
        user:           "username",
        password:       "password",
        connectString:  "localhost/XE",
        poolMax:        10,
        poolMin:        2
    },
    sqls = [],
    i    = 15;

while(i--) { sqls.push('SELECT SYSDATE FROM DUAL'); }

oracledb.createPool(cfg, function(err, pool) {
    var executeSql = function(sql, cb) {
        pool.getConnection(function(err, connection) {
            connection.execute(sql, [], function(err, results) {
                cb(null, results.rows[0][0]);
            });
        });
    };

    async.map(sqls, executeSql, function(err, results) {
        callback(null, results);
    });
});

Wynik: ORA-24418: Nie można otworzyć kolejnych sesji.

przykłady bez obsługi błędów oraz bez zwalniania połączenia do póli - dla przejrzystości
Wbudowana póla w driver natywny oracledb nie posiada aktualnie mechanizmów kolejki. Próba pobrania połączenia z póli w systuacji gdy wszystie połączenia pracują, kończy się niepowodzeniem.
