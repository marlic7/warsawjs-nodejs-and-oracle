Przykład 1b. Pobranie pojedynczej wartości (node-dal)

dal.selectOneValueSql('SELECT SYSDATE FROM DUAL', [],
                      function(err, result) {
    if(err) {
        cb(new Error(err));
        return;
    }
    cb(null, result);
});
dal.selectOneValue('DUAL', 'SYSDATE', [], function(err, result) {
    if(err) {
        cb(new Error(err));
        return;
    }
    cb(null, result);
});
w tej wersji pobieranie połączenia z póli i zwracanie jest transparentne
