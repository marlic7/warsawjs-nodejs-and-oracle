Przykład 1a. Pobranie pojedynczej wartości (oracledb)

pool.getConnection(function(err, connection) {
    if (err) {
        cb(new Error(err));
        return;
    }
    connection.execute('SELECT SYSDATE FROM DUAL',
                       [],
                       function(err, result) {
        if (err) {
            cb(new Error(err));
            connection.release(function(err) {
                if (err) {
                    cb(new Error(err), {only_log: true});
                }
            });
            return;
        }

        connection.release(function(err) {
            if (err) {
                cb(new Error(err));
                return;
            }

            cb(null, result.rows[0][0]);
        });
    });
});
    1. pobranie połączenia z póli (j. błąd to cb i return)
    2. wywołanie SQL (j. błąd to cb, zwolnienie połączenia do póli (z obsługą błędu) i return)
    3. zwolnienie połączenia do póli i przekazanie wyniku zapytania do głównego cb
